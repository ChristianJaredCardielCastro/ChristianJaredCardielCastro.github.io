a.idAsignatura, a.nombre AS nombre_asignatura FROM profesor p JOIN impartir i ON p.idProfesor = i.idProfesor JOIN asignatura a ON i.idAsignatura = a.idAsignatura ORDER BY
p.idProfesor, a.idAsignatura; Consulta #6.Obtener el listado de profesores y asignaturas que imparten


SELECT a.idAsignatura, a.nombre, COUNT(m.idAlumno) AS total_alumnos FROM asignatura a JOIN matricula m
ON a.idAsignatura = m.idAsignatura GROUP BY a.idAsignatura, a.nombre ORDER BY total_alumnos DESC LIMIT 1; Consulta #7.Mostrar la asignatura que tiene mas alumnos inscritos y su total.
